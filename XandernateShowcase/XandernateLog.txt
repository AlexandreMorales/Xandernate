SELECT pessoa.Id as Pessoa_Id, pessoa.Idade as Pessoa_Idade, pessoa.Altura as Pessoa_Altura, pessoa.Nome as Pessoa_Nome, pessoa.Peso as Pessoa_Peso, endereco.Id as Endereco_Id, endereco.Numero as Endereco_Numero, endereco.Complemento as Endereco_Complemento, endereco.Rua as Endereco_Rua, endereco.Cidade as Endereco_Cidade, endereco.Estado as Endereco_Estado, endereco.Cep as Endereco_Cep 
FROM Pessoa 
inner join Endereco endereco on  pessoa.EnderecoId = endereco.Id 
;
